    <script>

        const defaultFoods = ['火锅', '烧烤', '米线', '麻辣烫', '寿司', '炒饭', '面条', '汉堡'];

        let currentFoods = [...defaultFoods];

        

        const foodEmojis = {

            '火锅': '🍲',

            '烧烤': '🍢',

            '米线': '🍜',

            '麻辣烫': '🌶️',

            '寿司': '🍣',

            '炒饭': '🍚',

            '面条': '🍝',

            '汉堡': '🍔',

            '披萨': '🍕',

            '饺子': '🥟',

            '包子': '🥟',

            '沙拉': '🥗',

            '牛排': '🥩',

            '炸鸡': '🍗',

            '奶茶': '🧋',

            '咖啡': '☕',

            '蛋糕': '🍰'

        };

        

        const foodInput = document.getElementById('foodInput');

        const foodList = document.getElementById('foodList');

        const randomBtn = document.getElementById('randomBtn');

        const resetBtn = document.getElementById('resetBtn');

        const resultSection = document.getElementById('resultSection');

        const resultFood = document.getElementById('resultFood');

        const resultEmoji = document.getElementById('resultEmoji');

        

        function renderFoodList() {

            foodList.innerHTML = '';

            currentFoods.forEach((food, index) => {

                const tag = document.createElement('span');

                tag.className = 'food-tag inline-flex items-center gap-1 px-3 py-1.5 bg-gradient-to-r from-pink-100 to-yellow-100 text-pink-600 rounded-full text-sm font-medium cursor-pointer';

                tag.style.animationDelay = `${index * 0.05}s`;

                tag.innerHTML = `${getEmoji(food)} ${food}`;

                tag.title = '点击删除';

                tag.onclick = () => removeFood(index);

                foodList.appendChild(tag);

            });

        }

        

        function getEmoji(food) {

            return foodEmojis[food] || '🍽️';

        }

        

        function addFoods() {

            const input = foodInput.value.trim();

            if (!input) return;

            

            const newFoods = input.split(/[，,]/)

                .map(f => f.trim())

                .filter(f => f.length > 0 && !currentFoods.includes(f));

            

            if (newFoods.length > 0) {

                currentFoods = [...currentFoods, ...newFoods];

                renderFoodList();

                foodInput.value = '';

            }

        }

        

        function removeFood(index) {

            currentFoods.splice(index, 1);

            if (currentFoods.length === 0) {

                currentFoods = [...defaultFoods];

            }

            renderFoodList();

        }

        

        function resetFoods() {

            currentFoods = [...defaultFoods];

            foodInput.value = '';

            resultSection.classList.add('hidden');

            renderFoodList();

        }

        

        function getRandomFood() {

            if (currentFoods.length === 0) {

                currentFoods = [...defaultFoods];

            }

            return currentFoods[Math.floor(Math.random() * currentFoods.length)];

        }

        

        function animateAndShow() {

            resultSection.classList.add('hidden');

            randomBtn.classList.add('animate-bounce-soft');

            

            let count = 0;

            const maxCount = 15;

            const interval = setInterval(() => {

                const tempFood = getRandomFood();

                randomBtn.innerHTML = `<span class="inline-flex items-center gap-2"><span>🎲</span><span>${tempFood}</span><span>🎲</span></span>`;

                count++;

                

                if (count >= maxCount) {

                    clearInterval(interval);

                    const finalFood = getRandomFood();

                    showResult(finalFood);

                }

            }, 80);

        }

        

        function showResult(food) {

            randomBtn.classList.remove('animate-bounce-soft');

            randomBtn.innerHTML = `<span class="inline-flex items-center gap-2"><span>✨</span><span>开始随机！</span><span>✨</span></span>`;

            

            resultFood.textContent = food;

            resultEmoji.textContent = getEmoji(food);

            resultSection.classList.remove('hidden');

            resultSection.querySelector('div').classList.add('result-reveal');

            

            setTimeout(() => {

                resultSection.querySelector('div').classList.remove('result-reveal');

            }, 800);

        }

        

        foodInput.addEventListener('keypress', (e) => {

            if (e.key === 'Enter') {

                addFoods();

            }

        });

        

        foodInput.addEventListener('blur', addFoods);

        

        randomBtn.addEventListener('click', animateAndShow);

        

        resetBtn.addEventListener('click', resetFoods);

        

        renderFoodList();

    </script>
